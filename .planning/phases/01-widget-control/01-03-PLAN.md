**Change 1: Add autoMatchQuery helper function.** Add this as a method on StudioApp or as a local function accessible from the bind handler. Place it near `updateDataSummary`:

```javascript
/**
 * Attempt to find a saved query compatible with the new widget type.
 * Looks for a query with the same metricType as the current queryId.
 * @param {string} currentQueryId - The widget's current queryId
 * @param {string} currentSource - The widget's current source (e.g. 'gcp')
 * @returns {{ queryId: string, queryName: string } | null}
 */
autoMatchQuery(currentQueryId, currentSource) {
  if (!currentQueryId || !this.queries) return null;

  // Find the current query object
  const sourceQueries = this.queries[currentSource] || [];
  const currentQuery = sourceQueries.find(q => q.id === currentQueryId);
  if (!currentQuery || !currentQuery.metricType) return null;

  // The current query's metricType is still valid — it's the same metric, just a different viz
  // Return null to indicate "keep current query, no mismatch" — the metric type hasn't changed
  return null;
}
```

Note: Per RESEARCH.md, the auto-match heuristic in Phase 1 is intentionally simple: since the metricType doesn't change when the widget type changes (only the visualization changes), the existing queryId is almost always compatible. The auto-match only triggers a mismatch warning if the query cannot be found in the saved queries list (orphan query).

**Revised approach:** The mismatch warning should show when the current queryId exists but its metricType is known to be incompatible with the new widget type. In Phase 1, there is no shape-compatibility matrix, so the heuristic is:
- If queryId is empty: no warning (nothing to mismatch)
- If queryId exists and can be found in saved queries: no warning (assume compatible — same metric, different viz)
- If queryId exists but NOT found in saved queries: show warning (orphan query, can't verify compatibility)

```javascript
/**
 * Pure collision detection — mirrors _hasCollision in studio-canvas.js.
 * Used for unit testing snap logic without DOM/window dependency.
 */
export function hasCollision(dash, col, row, colSpan, rowSpan, excludeId) {
  return dash.widgets.some(function (w) {
    if (w.id === excludeId) return false;
    var wcs = w.position.colSpan || 1;
    var wrs = w.position.rowSpan || 1;
    var colOk = col < w.position.col + wcs && col + colSpan > w.position.col;
    var rowOk = row < w.position.row + wrs && row + rowSpan > w.position.row;
    return colOk && rowOk;
  });
}

/**
 * Pure snap-to-nearest — mirrors _snapToNearest in studio-canvas.js.
 * Used for unit testing snap logic without DOM/window dependency.
 */
export function snapToNearest(dash, desiredCol, desiredRow, colSpan, rowSpan, excludeId) {
  if (!hasCollision(dash, desiredCol, desiredRow, colSpan, rowSpan, excludeId)) {
    return { col: desiredCol, row: desiredRow };
  }
  var maxR = Math.max(dash.grid.columns, dash.grid.rows);
  for (var d = 1; d <= maxR; d++) {
    var candidates = [
      { col: desiredCol + d, row: desiredRow },
      { col: desiredCol,     row: desiredRow + d },
      { col: desiredCol - d, row: desiredRow },
      { col: desiredCol,     row: desiredRow - d },
    ];
    for (var i = 0; i < candidates.length; i++) {
      var c = Math.max(1, Math.min(dash.grid.columns - colSpan + 1, candidates[i].col));
      var r = Math.max(1, Math.min(dash.grid.rows    - rowSpan + 1, candidates[i].row));
      if (!hasCollision(dash, c, r, colSpan, rowSpan, excludeId)) {
        return { col: c, row: r };
      }
    }
  }
  return { col: desiredCol, row: desiredRow };
}
```
