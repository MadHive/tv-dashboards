    function computeThresholdState(value, thresholds, invert) {
      if (!thresholds) return 'ok';
      const { warning, critical } = thresholds;
      if (invert) {
        if (value <= critical) return 'crit';
        if (value <= warning)  return 'warn';
        return 'ok';
      }
      if (value >= critical) return 'crit';
      if (value >= warning)  return 'warn';
      return 'ok';
    }
